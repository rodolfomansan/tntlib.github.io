this.cookie = {
    get: function (cookieName) {
        'use strict';

        var results = document.cookie.match('(^|;) ?' + cookieName + '=([^;]*)(;|$)');
        if (results) {
            return (unescape(results[2]));
        } else {
            return null;
        }
    },

    set: function (name, value, exp_y, exp_m, exp_d, path, domain, secure) {
        'use strict';

        var cookie_string = name + "=" + window.escape(value);
        if (exp_y) {
            var expires = new Date(exp_y, exp_m, exp_d);
            cookie_string += '; expires=' + expires.toGMTString();
        }
        if (path) {
            cookie_string += '; path=' + window.escape(path);
        }
        if (domain) {
            cookie_string += '; domain=' + window.escape(domain);
        }
        if (secure) {
            cookie_string += '; secure';
        }
        document.cookie = cookie_string;
    }
};

// Adds a ref param to link or element
this.addRef = function (linkStr, ref) {
    'use strict';

    var link = document.createElement('a'), host, queries;

    link.href = linkStr;
    queries = link.href.match(/\?[^#?]*/g);

    if (queries) {
        link.search = "?";
        for (var i = 0; i < queries.length; i++) {
            link.search += queries[i].substring(1) + ((i < queries.length - 1) ? "&" : "");
        }
        if (link.hash) {
            link.hash = link.hash.replace(/\?[^#?]*/, "");
        }
    }

    if (link.href.indexOf("ref=") > -1) {
        return link.href.replace("ref=", "ref=" + ref + "|");
    }

    if (link.search) {
        return link.href.replace("?", "?ref=" + ref + "&");
    }

    if (link.href.indexOf("?") > -1) {
        var badQS = link.href.match(/\?([^#]*)/);
        link.href = link.href.replace(badQS[0], "");
        if (link.pathname.charAt(0) !== "/") { //IE wierdness
            host = (link.host.substr(-3) === ":80") ? link.hostname : link.host;
            return link.protocol + "//" + host + "/" + link.pathname + "?ref=" + ref + "&" + badQS[1] + link.hash;
        } else {
            return link.protocol + "//" + link.host + link.pathname + "?ref=" + ref + "&" + badQS[1] + link.hash;
        }
    }

    if (link.pathname.charAt(0) !== "/") { //IE wierdness
        host = (link.host.substr(-3) === ":80") ? link.hostname : link.host;
        return link.protocol + "//" + host + "/" + link.pathname + "?ref=" + ref + link.hash;
    } else {
        return link.protocol + "//" + link.host + link.pathname + "?ref=" + ref + link.hash;
    }
};

// Wraps a function and its args on a try catch with the bailout plugin integrated to it
this.wrapFunction = function (fn, msg) {
    'use strict';

    var _this = this;

    return function () {
        try {
            return fn.apply(_this, arguments);
        } catch (e) {
            _this.Error(msg, e);
        }
    };
};

// Wraps a function and its args with a timer, in order to keep track of run times
this.wrapTimer = function (name, fn) {
    'use strict';

    var _this = this;

    return function () {
        var args = Array.prototype.slice.call(arguments),
            startTime = new Date().getTime(),
            runTime, fnReturn;

        fnReturn = fn.apply(_this, args);

        runTime = new Date().getTime() - startTime;
        this.runtime.total += runTime;
        this.runtime.details.push(name + '(' + JSON.stringify(args).slice(1, -1) + ') >> ' + runTime + 'ms');

        return fnReturn;
    };
};

    this.checkNested = function (obj) {
    'use strict';
    var args = Array.prototype.slice.call(arguments);
    obj = args.shift();

    if (typeof obj === "undefined") {
        return false;
    } else {
        for (var i = 0; i < args.length; i++) {
            if (!obj.hasOwnProperty(args[i])) {
                return false;
            }
            obj = obj[args[i]];
        }
        return true;
    }
};
